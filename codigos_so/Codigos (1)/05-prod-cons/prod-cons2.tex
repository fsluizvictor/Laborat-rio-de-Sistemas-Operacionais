\begin{myslide}
  \heading{Produtor-Consumidor}
\subheading{Primeira tentativa}
\alltt{\normalsize
int c = 0;  /* Contador de posições ocupadas */
mutex_t lock_c; /* lock para o contador */

cond_t pos_vazia;   /* Para o produtor esperar */
cond_t pos_ocupada; /* Para o consumidor esperar */
}
\end{myslide}

\begin{myslide}
  \heading{Produtor-Consumidor}
\subheading{Primeira tentativa}
\alltt{\normalsize
int f = 0;
{\large\textbf{\textsf{Produtor:}}}
    mutex_lock(&lock_c);
    if (c == N) 
      cond_wait(&pos_vazia, &lock_c);
    f = (f+1)%N;
    buffer[f] = produz();
    c++;
    if (c == 1)
      cond_signal(&pos_ocupada);
    mutex_unlock(&lock_c);
}
\end{myslide}

\begin{myslide}
  \heading{Produtor-Consumidor}
\alltt{\normalsize
cond_t pos_vazia, pos_ocupada; mutex_t lock_v, lock_o;
int i = 0, f = 0, nv = N, no = 0;
{\large\textbf{\textsf{Produtor:}}}
    mutex_lock(&lock_v);
    if (nv == 0) cond_wait(&pos_vazia, &lock_v);
    nv--;
    mutex_unlock(&lock_v);
    f = (f+1)%N;
    buffer[f] = produz();
    mutex_lock(&lock_o);
    no++;
    cond_signal(&pos_ocupada);
    mutex_unlock(&lock_o);
}
\end{myslide}

\begin{myslide}
  \heading{Produtores-Consumidores}
  \subheading{Locks e variáveis de condição}
\alltt{
cond_t pos_vazia, pos_ocupada; 
mutex_t lock_v, lock_o;
int nv = N, no = 0;

mutex_t lock_i, lock_f;
int i = 0, f = 0;
}
\end{myslide}

\begin{myslide}
  \heading{Produtores-Consumidores}
\alltt{\normalsize
{\large\textbf{\textsf{Produtor:}}}
    item = produz();
    mutex_lock(&lock_v);
    if (nv == 0) cond_wait(&pos_vazia, &lock_v);
    nv--;
    mutex_unlock(&lock_v);
    mutex_lock(&lock_f);
    f = (f+1)%N;
    buffer[f] = item;
    mutex_unlock(&lock_f);
    mutex_lock(&lock_o);
    no++;
    cond_signal(&pos_ocupada);
    mutex_unlock(&lock_o);
}
\end{myslide}

\begin{myslide}
  \heading{Produtores-Consumidores}
\alltt{\normalsize
{\large\textbf{\textsf{Consumidor:}}}
    mutex_lock(&lock_o);
    if (no == 0) cond_wait(&pos_ocupada, &lock_o);
    no--;
    mutex_unlock(&lock_o);
    mutex_lock(&lock_i);
    i = (i+1)%N;
    item = buffer[i];
    mutex_unlock(&lock_i);
    mutex_lock(&lock_v);
    nv++;
    cond_signal(&pos_vazia);
    mutex_unlock(&lock_v);
    consome(item);
}
\end{myslide}
